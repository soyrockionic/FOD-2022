program Ej6P1;
Uses crt;
type

  celular = record
    codigo : integer;
    nombre : String[50];
    descrip : String[50];
    marca : String[50];
    precio : real;
    stockMin : integer;
    stockDisp : integer;
  end;

  archivo = file of celular;

procedure cargarArchivo(var arch : archivo);
var
  reg : celular;
  carga : text;
begin
  assign(carga, 'celulares.txt');
  rewrite(arch);
  reset(carga);
  while(not eof(carga)) do begin
    readln(carga, reg.codigo,reg.precio,reg.nombre,reg.marca);
    readln(carga, reg.stockMin,reg.stockDisp,reg.descrip);
    write(arch, reg);
  end;
  close(carga);
  close(arch);
end;

procedure stockMenorAlMinimo(var arch : archivo);
var
  reg : celular;
begin
  reset(arch);
  writeln('-------------------------');
  while(not eof(arch)) do begin
    read(arch, reg);
    if(reg.stockDisp < reg.stockMin) then
      writeln(reg.codigo,' ',reg.nombre,' ',reg.marca,' ',reg.precio:2:2,' ',reg.stockDisp);
  end;
  writeln('-------------------------');
  close(arch);
end;

procedure crearArchivoText(var arch : archivo);
var
  reg : celular;
  phone : text;
begin
  assign(phone, 'celular.txt');
  rewrite(phone);
  reset(arch);
  while(not eof(arch)) do begin
    read(arch, reg);
    write(phone, reg.codigo,' ',reg.nombre,' ',reg.marca,' ',reg.precio:2:2,' ',reg.stockDisp);
    writeln(phone, ' ');
  end;
  close(phone);
  close(arch);
end;

procedure contienenLosCharIngresados(var arch : archivo);
var
  reg : celular;
  cadena : String;
begin
  reset(arch);
  write('Ingrese la cadena de caracteres para verificar ');
  readln(cadena);
  while(not eof(arch)) do begin
    read(arch, reg);
    if(pos(cadena, reg.descrip) > 0) then
      writeln(reg.codigo,' ',reg.nombre,' ',reg.marca,' ',reg.descrip);
  end;
  close(arch);
end;

procedure leer(var reg : celular);
begin
  write('Ingrese codigo ');
  readln(reg.codigo);
  write('Ingrese nombre ');
  readln(reg.nombre);
  write('Ingrese descripcion ');
  readln(reg.descrip);
  write('Ingrese marca ');
  readln(reg.marca);
  write('Ingrese precio ');
  readln(reg.precio);
  write('Ingrese stock minimo ');
  readln(reg.stockMin);
  write('Ingrese stock disponible ');
  readln(reg.stockDisp);
end;

procedure agregarCelulares(var arch : archivo);
var
  reg : celular;
  cant, i : integer;
begin
  write('Ingrese la cantidad que desea agregar ');
  readln(cant);
  reset(arch);
  seek(arch, filesize(arch));
  for i := 1 to cant do begin
     leer(reg);
     write(arch, reg);
  end;
  close(arch);
end;

procedure modificarStock(var arch : archivo);
var
  reg : celular;
  n : String[50];
begin
  write('Ingrese  el nombre del celular que desa modificar su stock ');
  readln(n);
  reset(arch);
  while(not eof(arch)) and (reg.nombre <> n) do
    read(arch, reg);
  if(reg.nombre = n) then begin
    seek(arch, filepos(arch)-1);
    write('Ingrese el nuevo stock ');
    readln(reg.stockDisp);
    write(arch, reg);
  end;
  close(arch);
end;

procedure celularesSinStock(var arch : archivo);
var
  reg : celular;
  phone : text;
begin
  assign(phone, 'sinStock.txt');
  rewrite(phone);
  reset(arch);
  while(not eof(arch)) do begin
    read(arch, reg);
    if(reg.stockDisp = 0) then begin
      write(phone, reg.codigo,' ',reg.nombre,' ',reg.marca,' ',reg.descrip,' ',reg.precio:2:2,' ',reg.stockDisp);
      writeln(phone, ' ');
    end;
  end;
  close(arch);
  close(phone);
end;

var
  arch : archivo;
  car : char;
  n : String[50];
begin
  clrscr;

  write('Ingrese el nombre del archivo: ');
  readln(n);

  assign(arch, n);
  writeln('-------------------------');
  writeln('a : Cargar archivo');
  writeln('b : Celulares con stock menor al minimo');
  writeln('c : crear archivo de txt');
  writeln('d : celulares cuya descrip contiene los char ingresados');
  writeln('e : Agregar mas celulares al archivo');
  writeln('f : Modificar stock');
  writeln('g : Crear archivo con los celulares sin stock');
  readln(car);
  case car of
    'a' : cargarArchivo(arch);
    'b' : stockMenorAlMinimo(arch);
    'c' : crearArchivoText(arch);
    'd' : contienenLosCharIngresados(arch);
    'e' : agregarCelulares(arch);
    'f' : modificarStock(arch);
    'g' : celularesSinStock(arch);
  end;

  readln;
end.
