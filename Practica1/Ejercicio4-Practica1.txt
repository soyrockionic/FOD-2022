program Ej4P1;
Uses crt;
type

  empleado = record
    nro : integer;
    apellido : String[25];
    nombre : String[25];
    edad : integer;
    dni : integer;
  end;

  archivo = file of empleado;

procedure leer(var reg : empleado);
begin
  write('Ingrese apellido: ');
  readln(reg.apellido);
  if(reg.apellido <> 'fin') then begin
    write('Ingrese codigo: ');
    readln(reg.nro);
    write('Ingrese nombre: ');
    read(reg.nombre);
    write('Ingrese edad: ');
    readln(reg.edad);
    write('Ingrese dni: ');
    readln(reg.dni);
  end;
end;

procedure cargarArchivo(var arch : archivo);
var
  reg : empleado;
begin
  rewrite(arch);
  leer(reg);
  while(reg.apellido <> 'fin') do begin
    write(arch, reg);
    leer(reg);
  end;
  close(arch);
end;

procedure listaConElIngresado(var arch : archivo);
var
  reg : empleado;
  n : String;
begin
  write('Ingrese nombre para mostrar en pantalla ');
  readln(n);
  reset(arch);
  while(not eof(arch)) do begin
    read(arch, reg);
    if(reg.nombre = n) then
      writeln(reg.apellido,' ',reg.dni);
  end;
  close(arch);
end;

procedure todosLosEmpleados(var arch : archivo);
var
  reg : empleado;
begin
  reset(arch);
  while(not eof(arch)) do begin
    read(arch, reg);
    writeln(reg.nro,' ',reg.apellido,' ',reg.edad,' ',reg.dni);
  end;
  close(arch);
end;

procedure ProntoAJubilarse(var arch : archivo);
var
  reg : empleado;
begin
  reset(arch);
  while(not eof(arch)) do begin
    read(arch, reg);
    if(reg.edad > 70) then
      write(reg.nro,' ',reg.apellido);
  end;
  close(arch);
end;

procedure agregarNuevosEmpleados(var arch : archivo);
var
  reg : empleado;
  cant, i : integer;
begin
  write('Ingrese la cant de empleados que desea agregar ');
  readln(cant);
  reset(arch);
  seek(arch, filesize(arch));
  for i := 1 to cant do begin
    leer(reg);
    write(arch, reg);
  end;
  close(arch);
  todosLosEmpleados(arch);
end;

procedure modificarEdad(var arch : archivo);
var
  reg : empleado;
  cant, i, cod : integer;
begin
  reset(arch);
  write('Ingrese la cantidad de empledados que desea modificar la edad ');
  read(cant);
  for i := 1 to cant do begin
    write('Ingrese el codigo del empleado ');
    read(cod);
    while(not eof(arch)) and (reg.nro <> cod) do
      read(arch, reg);
    if(reg.nro = cod) then begin
      write('Ingrese la nueva edad ');
      readln(reg.edad);
      seek(arch, filepos(arch)-1);
      write(arch, reg);
    end;
    seek(arch, 0);
  end;
  close(arch);
  todosLosEmpleados(arch);
end;

procedure exportarArchivo(var arch : archivo);
var
  reg : empleado;
  carga : text;
begin
  assign(carga, 'todos_empleados.txt');
  rewrite(carga);
  reset(arch);
  while(not eof(arch)) do begin
    read(arch, reg);
    write(carga, reg.nro, ' ', reg.apellido, ' ', reg.edad, ' ', reg.dni);
    writeln(carga, ' ');
  end;
  close(carga);
  close(arch);
end;

procedure empleadosSinDni(var arch : archivo);
var
  carga : text;
  reg : empleado;
begin

  assign(carga, 'faltaDniEmpleado.txt');

  reset(arch);
  rewrite(carga);
  while(not eof(arch)) do begin
    read(arch, reg);
    if(reg.dni = 00) then begin
      write(carga, reg.nro,' ',reg.apellido,' ',reg.edad,' ',reg.dni);
      writeln(carga,' ');
    end;
  end;
  close(carga);
  close(arch);

end;

procedure recorrerArchivo(var arch : archivo);
var
  op : integer;
begin
  writeln();
  writeln('1 : lista de empleados con el nombre ingresado');
  writeln('2 : lista de todos los empleados');
  writeln('3 : lista de los empleados pronto a jubilarse');
  writeln('4 : agregar nuevos empleados');
  writeln('5 : modificar edad');
  writeln('6 : exportar a un txt');
  writeln('7 : empleadosSinDni');
  writeln();
  write('Ingrese que eperacion desea ');
  readln(op);
  case op of
    1 : listaConElIngresado(arch);
    2 : todosLosEmpleados(arch);
    3 : ProntoAJubilarse(arch);
    4 : agregarNuevosEmpleados(arch);
    5 : modificarEdad(arch);
    6 : exportarArchivo(arch);
    7 : empleadosSinDni(arch);
  end;
end;

var
  arch : archivo;
  n : String;
begin
  clrscr;

  write('Ingrese el nombre del archivo ');
  readln(n);

  assign(arch, n);

  cargarArchivo(arch);

  recorrerArchivo(arch);

  readln;
end.
