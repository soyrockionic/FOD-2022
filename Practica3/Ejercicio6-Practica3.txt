program Ej6_p3;
Uses crt, sysUtils;
const
  valorAlto = 32000;
type

  prenda = record
    cod_prenda : integer;
    descripcion : String[50];
    //colores : String[30];
    tipo_prenda : String[50];
    stock : integer;
    precio_unitario : real;
  end;

  maestro = file of prenda;
  detalle = file of integer;

procedure leer(var arch : maestro; var reg : prenda);
begin
  if(not eof(arch)) then
    read(arch,reg)
  else
    reg.cod_prenda := valorAlto;
end;

procedure compactar(var mae : maestro; var new_mae : maestro);
var
  reg : prenda;
begin
  assign(new_mae,'New_Prendas.dot');
  rewrite(new_mae);
  reset(mae);
  leer(mae,reg);
  while(reg.cod_prenda <> valorAlto) do begin
    if(reg.stock > 0) then
      write(new_mae,reg);
    leer(mae,reg);
  end;
  close(new_mae);
  close(mae);
end;

procedure informe(var arch : maestro);
var
  reg : prenda;
begin
  writeln('---------------------------------------');
  assign(arch,'Prendas.dot');
  reset(arch);
  leer(arch,reg);
  while(reg.cod_prenda <> valorAlto) do begin
    writeln(reg.cod_prenda,' ',reg.stock);
    leer(arch,reg);
  end;
  close(arch);
end;

procedure eliminar(var mae : maestro; cod : integer);
var
  reg : prenda;
begin
  reset(mae);
  leer(mae,reg);
  while(reg.cod_prenda <> cod) do
    leer(mae,reg);
  reg.stock := reg.stock * -1;
  seek(mae,filepos(mae)-1);
  write(mae,reg);
  close(mae);
end;

procedure actualizar(var mae : maestro; var det : maestro);
var
  reg : prenda;
  regD : prenda;
  new_mae : maestro;
begin
  reset(det);
  leer(det,regD);
  while(regD.cod_prenda <> valorAlto) do begin
    eliminar(mae,regD.cod_prenda);
    leer(det,regD);
  end;
  close(det);
  compactar(mae, new_mae);
  informe(mae);
  Erase(mae);
  Rename(new_mae,'Prendas.dot');
end;

var
  fecha : TDateTime;
  mae : maestro;
  det : maestro; reg : prenda;
begin
  clrscr;

  fecha := Date;
  writeln('Fecha de hoy: ',DateToStr(fecha),' Hora: ',FormatDateTime('HH:NN:SS',Time()));

  assign(mae,'Prendas.dot');
  assign(det,'Obsoletas.dot');

  informe(mae);

  actualizar(mae,det);

  informe(mae);

  readln;
end.
